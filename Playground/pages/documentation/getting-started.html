Getting Started with NUI

A complete guide to setting up a project with the NUI component library

Quick Start

The minimal setup to start using NUI in your project:

<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<meta name="viewport" content="width=device-width, initial-scale=1.0">
	<meta name="color-scheme" content="light dark">
	<!-- NUI Theme (REQUIRED) -->
	<link rel="stylesheet" href="NUI/css/nui-theme.css">
</head>
<body>
	<nui-button>
		<button type="button">Click Me</button>
	</nui-button>

	<script type="module" src="NUI/nui.js"></script>
</body>
</html>
Important: Include CSS in <head> and JS as ES Module to prevent Flash of Unstyled Content (FOUC).

Project Structure

A typical NUI project follows this structure:

your-project/
├── index.html              # Entry point
├── NUI/                    # NUI library (copy or submodule)
│   ├── nui.js              # Core library (ES Module)
│   ├── css/
│   │   ├── nui-theme.css           # REQUIRED - core theme
│   │   └── modules/                # Optional module CSS
│   │       ├── nui-menu.css
│   │       ├── nui-list.css
│   │       └── nui-syntax-highlight.css
│   ├── lib/modules/                  # Optional addon JS
│   │   ├── nui-animation.js
│   │   ├── nui-list.js
│   │   └── nui-menu.js
│   └── assets/
│       └── material-icons-sprite.svg
├── js/
│   └── main.js           # Your application code
└── pages/                # Page fragments (if using SPA routing)
    ├── index.html
    └── components/
        └── button.html

Two Ways to Use NUI

1. Simple: Standalone Components

Use NUI components on any HTML page without the SPA routing system:

<!DOCTYPE html>
<html lang="en">
<head>
	<link rel="stylesheet" href="NUI/css/nui-theme.css">
	<script type="module" src="NUI/nui.js"></script>
</head>
<body>
	<nui-button>
		<button type="button">Save</button>
	</nui-button>

	<nui-tabs>
		<menu>
			<button data-tab="info">Info</button>
			<button data-tab="settings">Settings</button>
		</menu>
		<section data-panel="info">
			<p>This is the info tab.</p>
		</section>
		<section data-panel="settings">
			<p>Settings go here.</p>
		</section>
	</nui-tabs>
</body>
</html>

2. Full: SPA Application

Use the built-in router for multi-page applications with hash-based navigation:

<!-- index.html -->
<!DOCTYPE html>
<html lang="en">
<head>
	<link rel="stylesheet" href="NUI/css/nui-theme.css">
	<link rel="stylesheet" href="NUI/css/modules/nui-list.css">
	<script type="module" src="js/main.js"></script>
</head>
<body>
	<nui-loading mode="bar"></nui-loading>
	<nui-app nui-vars-sidebar_width="15rem">
		<nui-top-nav>
			<header>
				<nui-button data-action="toggle-sidebar">
					<button type="button" aria-label="Toggle menu">☰</button>
				</nui-button>
				<h1>My App</h1>
			</header>
		</nui-top-nav>
		<nui-side-nav>
			<nui-link-list mode="fold"></nui-link-list>
		</nui-side-nav>
		<nui-content>
			<nui-main><!-- Pages load here --></nui-main>
		</nui-content>
	</nui-app>
</body>
</html>
 
// js/main.js
import { nui } from '../NUI/nui.js';
import '../NUI/lib/modules/nui-list.js';

// Navigation data
const navigationData = [
	{
		label: 'Pages',
		items: [
			{ label: 'Home', href: '#page=home' },
			{ label: 'About', href: '#page=about' }
		]
	}
];

// Load sidebar
document.querySelector('nui-link-list').loadData(navigationData);

// Enable SPA routing
nui.enableContentLoading({
	container: 'nui-main',
	navigation: 'nui-side-nav',
	basePath: 'pages',
	defaultPage: 'home'
});

Layout Modes

NUI supports two layout modes based on whether <nui-app> is present:

Use for: Desktop-style applications with sidebar navigation

<nui-app nui-vars-sidebar_width="15rem">
	<nui-top-nav><header><h1>My App</h1></header></nui-top-nav>
	<nui-side-nav><nav>...</nav></nui-side-nav>
	<nui-content><nui-main>...</nui-main></nui-content>
	<nui-app-footer><footer>...</footer></nui-app-footer>
</nui-app>

Creates CSS Grid layout with fixed regions, independent scrolling, and responsive sidebar collapse.

Use for: Traditional websites, documentation, blogs

<nui-top-nav><header><h1>My Site</h1></header></nui-top-nav>
<nui-content><main>...</main></nui-content>
<nui-app-footer><footer>...</footer></nui-app-footer>

Components flow naturally in document layout. No fixed positioning.

Component Patterns

Components follow two patterns - know which to use:

Declarative (HTML-First) - Most Components

Define structure in HTML, components enhance existing elements:

<!-- Button: wrap a button -->
<nui-button>
	<button type="button">Click Me</button>
</nui-button>

<!-- Tabs: use data-tab/data-panel attributes -->
<nui-tabs>
	<menu>
		<button data-tab="tab1">Tab 1</button>
		<button data-tab="tab2">Tab 2</button>
	</menu>
	<section data-panel="tab1">Content 1</section>
	<section data-panel="tab2">Content 2</section>
</nui-tabs>

<!-- Accordion: use details/summary -->
<nui-accordion>
	<details>
		<summary>Section 1</summary>
		<p>Content here</p>
	</details>
</nui-accordion>

Data-Driven (JavaScript) - Exceptions

Use factory methods for complex runtime structures:

// Dialog - system modal
await nui.components.dialog.confirm('Delete?', 'This cannot be undone');

// Banner - toast notification
nui.components.banner.show({ content: 'Saved!', priority: 'info', autoClose: 3000 });

// LinkList - navigation tree
const nav = nui.components.linkList.create([
	{ label: 'Home', href: '#home' },
	{ label: 'Settings', items: [
		{ label: 'Profile', href: '#profile' },
		{ label: 'Security', href: '#security' }
	]}
], { mode: 'fold' });
document.querySelector('nav').appendChild(nav);
Rule: Default to declarative HTML. Only use data-driven APIs when content is dynamic or complex.

SPA Routing & Architecture

NUI includes a flexible, hash-based routing system for building Single Page Applications (SPAs). The router acts as a state coordinator that translates URL hashes into DOM visibility.

Learn More: See the Architecture Patterns guide for a deep dive into the different ways to structure your application (Centralized vs. Fragment-Based vs. Hybrid).

Enable Routing

The simplest way to get started is using the built-in content loader, which fetches HTML fragments based on the URL:

nui.enableContentLoading({
	container: 'nui-main',              // Where content loads
	navigation: 'nui-side-nav',            // Auto-close sidebar on navigate
	basePath: 'pages',                    // Folder containing page fragments
	defaultPage: 'home'                   // Default page (no hash in URL)
});

Navigation Links

Use hash format #type=path/to/page. The router will automatically handle the navigation and history:

<a href="#page=components/button">Button Docs</a>
<a href="#feature=dashboard">Open Dashboard</a>

Features (JavaScript-Only Pages)

For complex applications, you can register pages built entirely in JavaScript without fetching HTML fragments:

// In main.js - register before enableContentLoading
nui.registerFeature('dashboard', (container, params) => {
	// Build UI programmatically
	container.innerHTML = `
		<header><h1>Dashboard</h1></header>
		<div class="stats">Loading...</div>
	`;

	// Optional: add show/hide lifecycle
	container.show = () => { console.log('Dashboard shown'); };
	container.hide = () => { console.log('Dashboard hidden'); };
});

// Navigate to feature: #feature=dashboard

Interaction Patterns

Event Delegation with data-action

Handle clicks without manual event listeners:

<!-- Define action in HTML -->
<nui-button data-action="save">
	<button type="button">Save</button>
</nui-button>

<nui-button data-action="toggle-theme">
	<button type="button">Toggle Theme</button>
</nui-button>
 
// Handle in JavaScript (event delegation)
document.addEventListener('click', (e) => {
	const el = e.target.closest('[data-action]');
	if (!el) return;

	const action = el.dataset.action;

	switch (action) {
		case 'save':
			console.log('Saving...');
			break;
		case 'toggle-theme':
			const current = document.documentElement.style.colorScheme || 'light';
			document.documentElement.style.colorScheme = current === 'dark' ? 'light' : 'dark';
			break;
		case 'toggle-sidebar':
			document.querySelector('nui-app')?.toggleSideNav();
			break;
	}
});

Custom Events

Components dispatch standard CustomEvents:

// Listen for component events
document.querySelector('nui-tabs').addEventListener('nui-tab-change', (e) => {
	console.log('Tab changed to:', e.detail.tab);
});

// Dialog results
document.querySelector('nui-dialog')?.addEventListener('nui-dialog-close', (e) => {
	console.log('Dialog closed, confirmed:', e.detail.confirmed);
});

CSS Variables

Customize appearance with CSS custom properties:

Layout Variables

/* On nui-app element */
nui-app {
	--nui-sidebar-width: 15rem;
	--nui-sidebar-force-breakpoint: 75rem;
	--nui-topnav-height: 3.5rem;
}

Theme Variables

/* Global theming */
:root {
	--nui-accent: #0066cc;
	--nui-accent-hover: #0052a3;
	--nui-radius: 6px;
	--nui-space: 1rem;
}

Set via attributes on nui-app:

<nui-app nui-vars-sidebar_width="20rem" nui-vars-accent="#e74c3c">

Preventing Flash of Unstyled Content

Add inline styles to hide content until components upgrade:

<head>
	<link rel="stylesheet" href="NUI/css/nui-theme.css">
	<style>
		body { margin: 0; overflow: hidden; }
		nui-app:not(.nui-ready) { display: none; }
		nui-loading:not(.active) { display: none; }
	</style>
	<script type="module" src="js/main.js"></script>
</head>
<body>
	<nui-loading mode="bar"></nui-loading>
	<nui-app>
		<!-- Content hidden until .nui-ready class added -->
	</nui-app>
</body>

Complete Example

A working application combining all patterns:

<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<meta name="viewport" content="width=device-width, initial-scale=1.0">
	<meta name="color-scheme" content="light dark">
	<title>My NUI App</title>

	<link rel="stylesheet" href="NUI/css/nui-theme.css">
	<link rel="stylesheet" href="NUI/css/modules/nui-list.css">

	<style>
		body { margin: 0; overflow: hidden; }
		nui-app:not(.nui-ready) { display: none; }
	</style>

	<script type="module" src="js/main.js"></script>
</head>
<body>
	<nui-loading mode="bar"></nui-loading>

	<nui-app nui-vars-sidebar_width="15rem">
		<nui-top-nav>
			<header>
				<layout>
					<item>
						<nui-button data-action="toggle-sidebar">
							<button type="button" aria-label="Toggle menu">☰</button>
						</nui-button>
					</item>
					<item><h1>My App</h1></item>
					<item>
						<nui-button data-action="toggle-theme">
							<button type="button" aria-label="Toggle theme">◐</button>
						</nui-button>
					</item>
				</layout>
			</header>
		</nui-top-nav>

		<nui-side-nav>
			<nui-link-list mode="fold"></nui-link-list>
		</nui-side-nav>

		<nui-content>
			<nui-main></nui-main>
		</nui-content>
	</nui-app>
</body>
</html>
 
// js/main.js
import { nui } from '../NUI/nui.js';
import '../NUI/lib/modules/nui-list.js';

// Navigation
const navigationData = [
	{
		label: 'Getting Started',
		items: [
			{ label: 'Introduction', href: '#page=introduction' },
			{ label: 'Button', href: '#page=button' }
		]
	}
];

document.querySelector('nui-link-list').loadData(navigationData);

// Action handling
document.addEventListener('click', (e) => {
	const action = e.target.closest('[data-action]')?.dataset.action;
	if (!action) return;

	switch (action) {
		case 'toggle-sidebar':
			document.querySelector('nui-app')?.toggleSideNav();
			break;
		case 'toggle-theme':
			const current = document.documentElement.style.colorScheme || 'light';
			document.documentElement.style.colorScheme = current === 'dark' ? 'light' : 'dark';
			break;
	}
});

// Enable routing
nui.enableContentLoading({
	container: 'nui-main',
	navigation: 'nui-side-nav',
	basePath: 'pages',
	defaultPage: 'introduction'
});
